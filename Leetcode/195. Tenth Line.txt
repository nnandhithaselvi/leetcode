import java.io.*;

public class Main {
    public static void main(String[] args) {
        try {
            BufferedReader br = new BufferedReader(new FileReader("file.txt"));
            String line;
            int count = 0;

            while ((line = br.readLine()) != null) {
                count++;
                if (count == 10) {
                    System.out.println(line);
                    break;
                }
            }

            br.close();
        } catch (Exception e) {
            System.out.println("");
        }
    }
}
